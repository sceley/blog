gulp自动化

前端
  • JAVASCRIPT

序言

gulp是前端开发过程中对代码进行构建的工具，是自动化项目的构建利器；它不仅能对网站资源进行优化，而且在开发过程中很多重复的任务能够使用正确的工具自动完成；使用它，我们不仅可以很愉快的编写代码，而且大大提高我们的工作效率。

gulp

入门指南

全局安装 gulp：

$ npm install --global gulp

作为项目的开发依赖（devDependencies）安装：

$ npm install --save-dev gulp

在项目根目录下创建一个名为 gulpfile.js 的文件：

let gulp = require('gulp');
gulp.task('default', function() {
  // 将你的默认的任务代码放在这
});

运行 gulp：

$ gulp

默认的名为 default 的任务（task）将会被运行，在这里，这个任务并未做任何事情。

gulpfile文件的配置

let gulp = require("gulp");
let path = require('path');
let babel = require("gulp-babel");
let autoprefixer = require("gulp-autoprefixer");
let browserSync = require("browser-sync");
let sass = require("gulp-ruby-sass");
let rename = require("gulp-rename");
let uglify = require("gulp-uglify");
let concat = require("gulp-concat");
let app = require("./server/app").app;
gulp.task('server', () => {
    app();
});
//编译es6成es5
//注： 一定要同时也安装npm install babel-preset-es2015 --save-dev 否则报错
gulp.task('js', () => {
    gulp.src('build/js/*.js')
    .pipe(concat("es5.js"))//把所有的文件整合到es5.js文件
    .pipe(babel({presets: ["es2015"]}))//转码es2015
    .pipe(uglify())//不能压缩es2015,必须转码后再压缩
    .pipe(rename("es.min.js"))//重命名文件
    .pipe(gulp.dest('dest/js'));
});
gulp.task("auto", () => {
    //监听变化的文件, 执行相应的任务
    gulp.watch(['build/js/*.js', 'build/css/*.css'], ['babel', 'fix_css']);
    //参数为监听的文件, 参数二为要执行的任务
});
//自动补全css3前缀兼容
gulp.task("fix_css", () => {
    gulp.src('build/css/*.css')
    .pipe(autoprefixer())
    .pipe(gulp.dest("dest/css"));
});
//注记. 创建sass文件时命名应该是.scss不能是.SCSS否则编译时会出错
gulp.task("sass", () => {
    sass('build/sass/*.scss')//编译.scss文件成.css文件
    .pipe(autoprefixer())//自动补全css3前缀兼容
    .pipe(gulp.dest("dest/sass"));
});
//浏览器自动更新
gulp.task("browser-sync", () => {
    browserSync.init({
        files: ['dest/css/*.css', 'server/*.html'],//监听变化的文件
        server:{
            baseDir: './',// 设置服务器的根目录
            index: "server/index.html"// 指定默认打开的文件
        },
        port: 8080// 指定访问服务器的端口号
    });
});
//默认执行gulp时执行的任务
gulp.task('default', ['server', "sass", "js", "auto", "fix_css", "browser-sync"]);
